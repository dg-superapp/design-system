```tsx
"use client";
import * as React from "react";
import { Menu } from "lucide-react";
import { cn } from "@/lib/utils";
import { Badge } from "@/components/ui/badge";

export interface AppHeaderProps extends React.HTMLAttributes<HTMLElement> {
  title?: string;
  leading?: React.ReactNode;
  trailing?: React.ReactNode;
  ariaLabel?: string;
  as?: "h1" | "h2";
}

function AppHeaderRoot({ title, leading, trailing, ariaLabel, as = "h1", className, children, ...rest }: AppHeaderProps) {
  // Children path overrides slots (escape hatch per D-06)
  if (children) {
    return (
      <header
        role="banner"
        aria-label={ariaLabel}
        className={cn(
          "app-header relative",
          "min-h-[var(--app-header-h)]",
          "grid grid-cols-[var(--app-header-icon)_1fr_auto] items-center",
          "bg-[image:var(--gradient-hero)] text-white",
          className,
        )}
        {...rest}
      >
        {children}
      </header>
    );
  }
  // Slots path (80% case — D-06)
  const TitleEl = as;
  return (
    <header
      role="banner"
      aria-label={ariaLabel}
      className={cn(
        "app-header relative",
        "min-h-[var(--app-header-h)]",                                            // Pitfall 3 — min-h not h
        "grid grid-cols-[var(--app-header-icon)_1fr_auto] items-center",          // D-07 intrinsic right col
        "bg-[image:var(--gradient-hero)] text-white",
        className,
      )}
      {...rest}
    >
      <div className="flex items-center justify-center">
        {leading ?? <AppHeaderIconButton ariaLabel="Open menu / បើកម៉ឺនុយ"><Menu size={22} /></AppHeaderIconButton>}
      </div>
      {title ? (
        <TitleEl
          data-testid="app-header-title"
          className="truncate px-[var(--space-2)] text-base font-semibold"
        >
          {title}
        </TitleEl>
      ) : <span aria-hidden="true" />}
      <div className="flex items-center justify-end gap-[var(--space-2)] pr-[var(--space-2)]">
        {trailing}
      </div>
    </header>
  );
}

// Dot-namespace slot wrappers (D-03 + D-06 escape hatch)
function AppHeaderTitle({ children, as = "h1" }: { children: React.ReactNode; as?: "h1" | "h2" }) {
  const TitleEl = as;
  return (
    <TitleEl data-testid="app-header-title" className="truncate px-[var(--space-2)] text-base font-semibold">
      {children}
    </TitleEl>
  );
}
function AppHeaderLeading({ children }: { children: React.ReactNode }) {
  return <div className="flex items-center justify-center">{children}</div>;
}
function AppHeaderTrailing({ children }: { children: React.ReactNode }) {
  return <div className="flex items-center justify-end gap-[var(--space-2)] pr-[var(--space-2)]">{children}</div>;
}

// Internal IconButton — NOT exported as registry public surface, but exposed via dot-namespace
// for the badge variant. Applies data-app-header-icon for the focus-visible hook (Pitfall 9).
export interface AppHeaderIconButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  ariaLabel: string;
  badge?: "dot" | number;
}
function AppHeaderIconButton({ ariaLabel, badge, className, children, ...rest }: AppHeaderIconButtonProps) {
  return (
    <button
      type="button"
      data-app-header-icon
      aria-label={ariaLabel}
      className={cn(
        "relative inline-flex items-center justify-center",
        "h-[var(--app-header-icon)] w-[var(--app-header-icon)]",                  // 44×44 hit area (R5.1)
        "rounded-[var(--radius-sm)] bg-transparent text-white",
        "hover:bg-white/10 transition-colors duration-[var(--dur-fast)]",
        "focus:outline-none focus-visible:[box-shadow:var(--ring-on-navy)]",      // Pitfall 9 — white ring on navy
        "disabled:opacity-50 disabled:cursor-not-allowed disabled:pointer-events-none",
        className,
      )}
      {...rest}
    >
      <span aria-hidden="true" className="inline-flex items-center justify-center [&>svg]:h-[var(--app-header-glyph)] [&>svg]:w-[var(--app-header-glyph)]">
        {children}
      </span>
      {badge === "dot" && (
        <span
          aria-hidden="true"
          className="absolute top-[6px] right-[6px] h-[8px] w-[8px] rounded-full bg-[hsl(var(--red-600))] ring-2 ring-[hsl(var(--blue-900))]"
        />
      )}
      {typeof badge === "number" && badge > 0 && (
        <span
          aria-live="polite"
          aria-atomic="true"
          aria-label={`${badge} unread / ${badge} មិនបានអាន`}
          className="absolute top-[2px] right-[2px]"
        >
          <Badge
            className={cn(
              "h-[18px] min-w-[18px] px-[4px] py-0 text-[10px] font-medium leading-none",
              "bg-[hsl(var(--red-700))] text-white border-2 border-[hsl(var(--blue-900))]",
            )}
          >
            {badge}
          </Badge>
        </span>
      )}
    </button>
  );
}

export const AppHeader = Object.assign(AppHeaderRoot, {
  Title: AppHeaderTitle,
  Leading: AppHeaderLeading,
  Trailing: AppHeaderTrailing,
  IconButton: AppHeaderIconButton,
});
```

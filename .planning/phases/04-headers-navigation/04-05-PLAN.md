```tsx
"use client";
import * as React from "react";
import { Check } from "lucide-react";
import { cn } from "@/lib/utils";

export type StepState = "pending" | "active" | "done";
export type StepData = { label: string; state: StepState };

export interface StepIndicatorProps {
  steps?: ReadonlyArray<StepData>;
  ariaLabel?: string;
  children?: React.ReactNode;
  className?: string;
}

const StepContext = React.createContext<{ totalSteps: number } | null>(null);

function StepIndicatorRoot({ steps, ariaLabel, children, className }: StepIndicatorProps) {
  // D-08: array path renders compound internally
  const stepNodes = steps
    ? steps.map((s, i) => <StepIndicatorStep key={i} index={i + 1} {...s} />)
    : children;
  const totalSteps = steps?.length ?? React.Children.count(children);
  return (
    <StepContext.Provider value={{ totalSteps }}>
      <ol
        aria-label={ariaLabel}
        className={cn("relative flex items-start gap-[var(--space-4)]", className)}
      >
        {stepNodes}
      </ol>
    </StepContext.Provider>
  );
}

interface StepIndicatorStepProps extends StepData {
  index?: number;
}

function StepIndicatorStep({ label, state, index }: StepIndicatorStepProps) {
  const ctx = React.useContext(StepContext);
  const total = ctx?.totalSteps ?? 0;
  const stateLabel = state === "done" ? "completed / បានបញ្ចប់" : state === "active" ? "active / សកម្ម" : "pending / កំពុងរង់ចាំ";
  const isLast = index != null && index === total;
  const showConnector = !isLast;

  return (
    <li
      aria-current={state === "active" ? "step" : undefined}
      aria-label={`${index ?? ""}. ${label}, ${stateLabel}`}
      className="relative flex flex-col items-center flex-1 min-w-0"
    >
      <div
        className={cn(
          "relative flex items-center justify-center",
          "h-[28px] w-[28px] rounded-full text-base font-semibold",
          state === "pending" && "bg-[hsl(var(--bg-surface-2))] text-muted-foreground",
          state === "active" && "bg-brand text-brand-foreground ring-2 ring-[hsl(var(--brand)/0.25)] ring-offset-2 ring-offset-background",
          state === "done" && "bg-brand text-brand-foreground",
          "transition-colors duration-[var(--dur-base)]",
        )}
      >
        {state === "done" ? <Check aria-hidden="true" className="h-[16px] w-[16px]" /> : (index ?? "")}
      </div>
      <div className="mt-[var(--space-2)] text-xs font-medium text-foreground text-center max-w-[88px] truncate">
        {label}
      </div>
      {showConnector && (
        <span
          aria-hidden="true"
          className={cn(
            "absolute top-[14px] left-[calc(50%+14px+var(--space-2))] right-[calc(-50%+14px+var(--space-2))]",
            "h-[2px]",
            state === "done" ? "bg-brand" : "bg-[hsl(var(--bg-surface-2))]",
          )}
        />
      )}
    </li>
  );
}

export const StepIndicator = Object.assign(StepIndicatorRoot, {
  Step: StepIndicatorStep,
});
```

Connector geometry rationale (UI-SPEC §Tokens "StepIndicator connector render strategy"):
- `top: 14px` is `28px / 2 - 1px` (vertically centered against the 28px circle, minus half the 2px bar height).
- `left: calc(50% + 14px + var(--space-2))` starts the bar at the right edge of the current circle (center + half-circle-width + the gap).
- `right: calc(-50% + 14px + var(--space-2))` ends at the left edge of the next circle.
- These exact values align the bar between consecutive circles using the parent `<li class="flex-1">` flex distribution.

Notes:
- Connector color is determined by the CURRENT step's state (`bg-brand` if done, `bg-[hsl(var(--bg-surface-2))]` if pending/active) per UI-SPEC §Tokens "active circle's preceding bar is bg-brand; the bar leading OUT is pending."
  - This means: for state=done → outgoing bar is brand (the path INTO the next step is "completed").
  - For state=active or pending → outgoing bar is surface-2.
  - The bar leading INTO an active step is brand because the previous step was done — that's automatic with this rule.
- DO use `<ol>` for ordered list semantics (UI-SPEC §ARIA).
- DO NOT use `dark:*` (D-06).
- DO NOT add `usePathname` (D-15).
- DO NOT make steps clickable in v1 — UI-SPEC matrix marks "n/a (each step is data-driven, not interactive)".

"use client";
import { StepIndicator, type StepData } from "../../../../../registry/step-indicator/step-indicator";
import type { PreviewRendererProps } from ".";

const FIXTURE = [
  { label: "បញ្ជាក់អត្តសញ្ញាណ" },     // Identify
  { label: "ផ្ទៀងផ្ទាត់" },             // Verify
  { label: "បង់ប្រាក់" },               // Pay
  { label: "បញ្ចប់" },                  // Done
  { label: "មើលសេចក្តីសង្ខេប" },      // Review
];

export function StepIndicatorPreview({ state }: PreviewRendererProps) {
  const stepCount = Math.max(3, Math.min(5, Number(state.stepCount ?? "4")));
  const activeIndex = Math.max(0, Math.min(stepCount - 1, Number(state.activeIndex ?? 1)));
  const steps: StepData[] = FIXTURE.slice(0, stepCount).map((s, i) => ({
    label: s.label,
    state: i < activeIndex ? "done" : i === activeIndex ? "active" : "pending",
  }));

  return (
    <div className="w-full max-w-[412px] mx-auto p-[var(--space-6)] bg-card" lang="km">
      <StepIndicator steps={steps} ariaLabel="Application progress / វឌ្ឍនភាពពាក្យសុំ" />
    </div>
  );
}

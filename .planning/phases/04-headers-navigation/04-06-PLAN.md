```tsx
"use client";
import * as React from "react";
import { cn } from "@/lib/utils";

type SlideData = { title: string; body?: string; cta?: React.ReactNode };

export interface HeroBannerProps {
  activeIndex: number;
  onIndexChange: (i: number) => void;
  slides?: ReadonlyArray<SlideData>;
  ariaLabel?: string;
  children?: React.ReactNode;
  className?: string;
}

const HeroBannerCtx = React.createContext<{ activeIndex: number; total: number } | null>(null);

function HeroBannerRoot({ activeIndex, onIndexChange, slides, ariaLabel, children, className }: HeroBannerProps) {
  // D-11: array path renders compound internally
  const slideNodes = slides
    ? slides.map((s, i) => (
        <HeroBannerSlide key={i} index={i + 1}>
          <div className="flex flex-col gap-[var(--space-2)]">
            <h2 className="text-xl font-semibold">{s.title}</h2>
            {s.body && <p className="text-sm font-normal opacity-90">{s.body}</p>}
            {s.cta && <div className="mt-[var(--space-3)]">{s.cta}</div>}
          </div>
        </HeroBannerSlide>
      ))
    : children;
  const total = slides?.length ?? React.Children.count(children);

  return (
    <HeroBannerCtx.Provider value={{ activeIndex, total }}>
      <section
        role="region"
        aria-roledescription="carousel"
        aria-label={ariaLabel}
        className={cn(
          "relative rounded-[var(--radius-lg)] overflow-hidden",
          "bg-[image:var(--gradient-hero)] text-white",
          "p-[var(--space-6)] min-h-[176px]",
          className,
        )}
      >
        {/* Stipple overlay (D-04 inline, D-12 token path) */}
        <div
          aria-hidden="true"
          className="absolute inset-0 pointer-events-none bg-[image:var(--pattern-stipple)] bg-[length:12px_12px]"
        />
        {/* Slide track (consumer-driven activeIndex per D-10) */}
        <div
          className="relative flex transition-transform duration-[var(--dur-base)]"
          style={{ transform: `translateX(-${activeIndex * 100}%)` }}
        >
          {slideNodes}
        </div>
        {/* Carousel dots (Gap #7) */}
        {total > 0 && (
          <HeroBannerDots total={total} active={activeIndex} onSelect={onIndexChange} />
        )}
      </section>
    </HeroBannerCtx.Provider>
  );
}

interface HeroBannerSlideProps {
  children: React.ReactNode;
  index?: number;     // 1-based, populated by Root (D-11) or consumer (compound path)
}

function HeroBannerSlide({ children, index }: HeroBannerSlideProps) {
  const ctx = React.useContext(HeroBannerCtx);
  const i0 = (index ?? 1) - 1;
  const isActive = ctx?.activeIndex === i0;
  const total = ctx?.total ?? 1;
  return (
    <div
      role="group"
      aria-roledescription="slide"
      aria-label={`Slide ${index ?? 1} of ${total}`}
      aria-hidden={!isActive}
      // @ts-expect-error — `inert` boolean attribute (React 19 supports it natively; A6 verified)
      inert={!isActive ? "" : undefined}
      tabIndex={isActive ? 0 : -1}
      className="relative shrink-0 grow-0 basis-full"
    >
      {children}
    </div>
  );
}

function HeroBannerCard({ children }: { children: React.ReactNode }) {
  // Pitfall 5: opaque-by-design. Use bg-card (semantic — dark mode flips automatically via theme.css).
  // No transparency — stipple must NOT bleed through (D-13 + Pitfall 5).
  return (
    <div className="rounded-[var(--radius-md)] bg-card text-card-foreground p-[var(--space-4)] shadow-[var(--shadow-2)]">
      {children}
    </div>
  );
}

function HeroBannerDots({ total, active, onSelect }: { total: number; active: number; onSelect: (i: number) => void }) {
  // Gap #7: 3 visible max — windowed render. For total ≤ 3, render all; for total > 3,
  // show dots [active-1, active, active+1] clamped to [0, total-1].
  const visibleStart = total <= 3 ? 0 : Math.max(0, Math.min(total - 3, active - 1));
  const visibleEnd = total <= 3 ? total : visibleStart + 3;

  return (
    <div
      aria-hidden="true"
      className="absolute bottom-[var(--space-4)] left-1/2 -translate-x-1/2 flex gap-[var(--space-2)]"
    >
      {Array.from({ length: total }).map((_, i) => {
        const inWindow = i >= visibleStart && i < visibleEnd;
        const isActive = i === active;
        return (
          <button
            key={i}
            type="button"
            tabIndex={-1}
            onClick={() => onSelect(i)}
            className={cn(
              "h-[8px] rounded-full transition-all duration-[var(--dur-base)]",
              !inWindow && "hidden",
              isActive ? "w-[18px] bg-white" : "w-[8px] bg-white/40",
            )}
          />
        );
      })}
    </div>
  );
}

export const HeroBanner = Object.assign(HeroBannerRoot, {
  Slide: HeroBannerSlide,
  Card: HeroBannerCard,
});
```

Notes:
- DO NOT use `useState` to track activeIndex anywhere — D-10 forbids. The acceptance criterion `grep -c "useState" registry/hero-banner/hero-banner.tsx` returns 0.
- DO NOT use `dark:*` (D-06).
- DO NOT add swipe handlers / keyboard handlers (D-10).
- DO use `inert` as a JSX attribute. React 19 supports it as a boolean attribute; the `@ts-expect-error` comment covers TypeScript not yet recognizing it on `HTMLAttributes` until `@types/react@19.x` ships the typing. If `inert` does NOT type-check after upgrade, fall back to `pointerEvents: 'none'` inline style + `aria-hidden` (A6 fallback). Document the fallback in 04-06-SUMMARY.md.

- Bg gradient + stipple stack: gradient on `<section>`, stipple on `<div absolute inset-0>`, content on `<div relative>` above. Stipple z-index < content z-index (default `position: relative` lifts content per stacking context).
